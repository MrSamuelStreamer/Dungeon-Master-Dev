namespace = dmdebug

#
# Toggles the flag to show/hide the offmap empire
# Offmap empires check display triggers once a month
#

character_event = {
	id = dmdebug.999
	hide_window = yes
	is_triggered_only = yes
	offmap = allow
	immediate = {
		if = {
			limit = {
				has_character_flag = show_dm_offmap
			}
			clr_character_flag = show_dm_offmap
		}
		else = {
			set_character_flag = show_dm_offmap
		}
	}
}
character_event = {
	id = dmdebug.998
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		add_character_modifier = {
			name = dm_bovid19
			duration = -1
		}
		culture = dm_cow_culture
		set_graphical_culture = dm_cow_culture
		if = {
			limit = {
				is_adult = yes
			}
			set_max_milk = yes
		}
		any_realm_character = {
			add_character_modifier = {
				name = dm_bovid19
				duration = -1
			}
			culture = dm_cow_culture
			set_graphical_culture = dm_cow_culture
			if = {
				limit = {
					is_adult = yes
				}
				set_max_milk = yes
			}
		}
	}
}
character_event = {
	id = dmdebug.997
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		location = {
			activate_disease = dm_bovid19_disease
			spawn_disease = dm_bovid19_disease
		}
	}
}
character_event = {
	id = dmdebug.996
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		create_character = {
			culture = dm_cow_culture
		}
	}
}

character_event = {
	id = dmdebug.995
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		activate_title = {
			title = k_dungeon
			status = no
		}
		any_character = {
			limit = {
				trait = dungeon_master
				NOR = {
					is_offmap_governor = yes
					is_offmap_ruler = yes
				}
				death = {
					death_reason = death_missing
				}
			}
		}
	}
}
