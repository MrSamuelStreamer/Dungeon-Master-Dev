character_event = {
	id = DMCore.999
	hide_window = yes
	is_triggered_only = yes # on_yearly_pulse
	offmap = allow
	trigger = {
		trait = dungeon_master
		is_offmap_governor = yes
		NOT = { has_global_flag = dungeon_master_working }
		NOT = {
			has_game_rule = {
				name = dungeon_master_frequency
				value = dm_freq_none
			}
		}
	}
	immediate = {
		set_global_flag = dungeon_master_working
		if = {
			limit = {
				has_game_rule = {
					name = dungeon_master_frequency
					value = dm_freq_5
				}
			}
			province_event = { id = DMCore.998 years = 4 random = 365 } # Four years minumum, five years maximum.
		}
		else_if = {
			limit = {
				has_game_rule = {
					name = dungeon_master_frequency
					value = dm_freq_30
				}
			}
			province_event = { id = DMCore.998 years = 29 random = 365 } # 29 years minumum, 30 years maximum.
		}
		else_if = {
			limit = {
				has_game_rule = {
					name = dungeon_master_frequency
					value = dm_freq_true_random
				}
			}
			province_event = { id = DMCore.998 years = 1 random = 3650 } # One year minumum, 10 years maximum.
		}
		else = {
			province_event = { id = DMCore.998 years = 14 random = 365 } # 14 years minumum, 15 years maximum.
		}
	}
}
