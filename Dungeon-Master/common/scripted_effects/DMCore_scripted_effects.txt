		limit = {
			trait = dungeon_master
			is_offmap_ruler = no
			e_dungeon_master_governor = {
				owner = {
					NOT = { character = PREVPREV }
				}
			}
		}
		remove_trait = dungeon_master

		death = {

		}
	}
	dm_gov_create = yes

duplicate_check = {			#holds events in storage so they can't be reused too soon, called from DMCore.1
	if = {
		limit = {
			has_game_rule = {
				name = dungeon_master_cycle
				value = dm_cycle_double
			}
		}
		if = {
			limit = {
				check_variable = { which = global_tombola_var value = 12 }
			}
			flag_clear = yes
		}
	}
	else_if = {
		limit = {
			has_game_rule = {
				name = dungeon_master_cycle
				value = dm_cycle_half
			}
		}
		if = {
			limit = {
				check_variable = { which = global_tombola_var value = 3 }
			}
			flag_clear = yes
		}
	}
	else_if = {
		limit = {
			has_game_rule = {
				name = dungeon_master_cycle
				value = dm_cycle_half
			}
		}
		if = {
			limit = {
				check_variable = { which = global_tombola_var value = 6 }
			}
			flag_clear = yes
		}
	}
}
