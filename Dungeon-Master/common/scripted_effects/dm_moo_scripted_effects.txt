dm_use_minor_milk = {
	subtract_variable = {
		which = current_milk
		value = 10
	}
}
dm_use_moderate_milk = {
	subtract_variable = {
		which = current_milk
		value = 25
	}
}
dm_use_major_milk = {
	subtract_variable = {
		which = current_milk
		value = 50
	}
}

yearly_milk_increase = {
	set_variable = {
		which = milk_increase
		which = max_milk
	}
	divide_variable = {
		which = milk_increase
		value = 10
	}
	change_variable = {
		which = current_milk
		which = milk_increase
	}
	
	if = {
		limit = {
			check_variable = {
				which = current_milk
				which >= max_milk
			}
		}
		set_variable = {
			which = current_milk
			which = max_milk
		}
	}
}

set_max_milk = {
	set_variable = {
		which = max_milk
		value = 0
	}
	
	if = {
		limit = {
			is_female = yes
		}
		change_variable = {
			which = max_milk
			value = 50
		}
	}
	else = {
		change_variable = {
			which = max_milk
			value = 25
		}
	}
	if = {
		limit = {
			dm_is_true_cow = yes
		}
		multiply_variable = {
			which = max_milk
			value = 1.2
		}
	}
	
	if = {
		limit = {
			NOT = {
				has_character_flag = dm_milk_set
			}
		}
		set_variable = {
			which = starting_milk
			which = max_milk
		}
		divide_variable = {
			which = starting_milk
			value = 2
		}
		set_variable = {
			which = current_milk
			which = starting_milk
		}
		set_character_flag = dm_milk_set
	}
	if = {
		limit = {
			NOT = {
				has_character_modifier = dm_milk_stores
			}
		}
		add_character_modifier = {
			name = dm_milk_stores
			duration = -1
		}
	}
}
